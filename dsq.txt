    local function IsAlive(plr)
        return plr and plr.Character and plr.Character:FindFirstChild("Humanoid") and plr.Character.Humanoid.Health > 0
    end
    
    local function findMurderer()
        local Players = game:GetService("Players")
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= Players.LocalPlayer then
                local hasKnife = p.Backpack:FindFirstChild("Knife") or (p.Character and p.Character:FindFirstChild("Knife"))
                if hasKnife and IsAlive(p) then
                    return p
                end
            end
        end
        return nil
    end
    
    local function findSheriff()
        local Players = game:GetService("Players")
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= Players.LocalPlayer then
                local hasGun = p.Backpack:FindFirstChild("Gun") or (p.Character and p.Character:FindFirstChild("Gun"))
                if hasGun and IsAlive(p) then
                    return p
                end
            end
        end
        return nil
    end
    
    local function findGun()
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("BasePart") and obj.Name == "GunDrop" then
                return obj
            end
        end
        return nil
    end
    
    local function findCoins()
        local coins = {}
        for _, container in pairs(workspace:GetDescendants()) do
            if container:IsA("Model") and container.Name == "CoinContainer" then
                for _, coinServer in pairs(container:GetChildren()) do
                    if coinServer.Name == "Coin_Server" then
                        local coinVisual = coinServer:FindFirstChild("CoinVisual")
                        if coinVisual then
                            local mainCoin = coinVisual:FindFirstChild("MainCoin")
                            if mainCoin and mainCoin:IsA("MeshPart") then
                                table.insert(coins, mainCoin)
                            end
                        end
                    end
                end
            end
        end
        return coins
    end
    
    local function GetMurdererFromRemote()
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local getPlayerDataRemote = ReplicatedStorage:FindFirstChild("GetPlayerData", true)
        if getPlayerDataRemote then
            local success, roles = pcall(function()
                return getPlayerDataRemote:InvokeServer()
            end)
            if success and roles then
                for name, data in pairs(roles) do
                    if data.Role == "Murderer" then
                        return game:GetService("Players"):FindFirstChild(name)
                    end
                end
            end
        end
        return findMurderer()
    end
    
    local function ShootMurderer()
        local player = game.Players.LocalPlayer
        local murderer = GetMurdererFromRemote()
        if not murderer or not murderer.Character then return false end
        
        local gun = player.Character and player.Character:FindFirstChild("Gun")
        if not gun and player.Backpack:FindFirstChild("Gun") then
            gun = player.Backpack:FindFirstChild("Gun")
            gun.Parent = player.Character
            task.wait(0.1)
            gun = player.Character:FindFirstChild("Gun")
        end
        
        if gun and gun:FindFirstChild("KnifeLocal") then
            local targetPart = murderer.Character:FindFirstChild("HumanoidRootPart") or murderer.Character:FindFirstChild("Head")
            if targetPart then
                local targetPos = targetPart.Position
                local mHum = murderer.Character:FindFirstChild("Humanoid")
                if _G.ShootPrediction ~= 0 and mHum then
                    local moveDir = mHum.MoveDirection
                    if moveDir.Magnitude > 0 then
                        local finalOffset = _G.ShootPrediction + (_G.PingMultiplier * 0.5)
                        targetPos = targetPos + (moveDir.Unit * finalOffset)
                    end
                end
                pcall(function()
                    gun.KnifeLocal.CreateBeam.RemoteFunction:InvokeServer(1, targetPos, "AH2")
                end)
                return true
            end
        end
        return false
    end

    local function setupESPDrawings()
        if espLoops.espRender then espLoops.espRender:Disconnect() end
        
        espLoops.espRender = game:GetService("RunService").RenderStepped:Connect(function()
            local camera = workspace.CurrentCamera
            if not camera then return end
            local screenCenter = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
            local players = game:GetService("Players"):GetPlayers()
            local localPlayer = game.Players.LocalPlayer
            
            -- Clean up old drawings for disconnected players
            for player, drawings in pairs(espDrawings) do
                if not player or not player.Parent then
                    for _, drawing in pairs(drawings) do
                        if drawing and drawing.Remove then pcall(drawing.Remove, drawing) end
                    end
                    espDrawings[player] = nil
                end
            end
            
            for _, plr in pairs(players) do
                if plr ~= localPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") and IsAlive(plr) then
                    local hrp = plr.Character.HumanoidRootPart
                    local role = nil
                    if findMurderer() == plr then role = "Murderer"
                    elseif findSheriff() == plr then role = "Sheriff"
                    else role = "Innocent" end
                    
                    local espConfig = nil
                    if role == "Murderer" then espConfig = _G.MurdererESP
                    elseif role == "Sheriff" then espConfig = _G.SheriffESP
                    else espConfig = _G.InnocentESP end
                    
                    if espConfig and (espConfig.box or espConfig.name or espConfig.tracer or espConfig.distance or espConfig.highlight) then
                        if not espDrawings[plr] then
                            espDrawings[plr] = {
                                box = Drawing.new("Square"),
                                name = Drawing.new("Text"),
                                tracer = Drawing.new("Line"),
                                distance = Drawing.new("Text"),
                                boxLines = {}
                            }
                            espDrawings[plr].box.Thickness = 2
                            espDrawings[plr].box.Filled = false
                            espDrawings[plr].name.Size = 14
                            espDrawings[plr].name.Center = true
                            espDrawings[plr].name.Outline = true
                            espDrawings[plr].tracer.Thickness = 1
                            espDrawings[plr].distance.Size = 12
                            espDrawings[plr].distance.Center = true
                            espDrawings[plr].distance.Outline = true
                        end
                        
                        local esp = espDrawings[plr]
                        local vec, onScreen = camera:WorldToViewportPoint(hrp.Position)
                        local color = (role == "Murderer" and Color3.fromRGB(255, 0, 0)) or (role == "Sheriff" and Color3.fromRGB(0, 0, 255)) or Color3.fromRGB(0, 255, 0)
                        
                        if onScreen then
                            local topY = camera:WorldToViewportPoint(hrp.Position + Vector3.new(0, 3, 0)).Y
                            local bottomY = camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0)).Y
                            local sizeY = math.abs(bottomY - topY)
                            local sizeX = sizeY / 1.5
                            
                            if espConfig.box then
                                if espConfig.box3d then
                                    -- 3D Box would be complex, using 2D for simplicity in merged version
                                    esp.box.Visible = true
                                    esp.box.Size = Vector2.new(sizeX, sizeY)
                                    esp.box.Position = Vector2.new(vec.X - sizeX/2, vec.Y - sizeY/2)
                                    esp.box.Color = color
                                else
                                    esp.box.Visible = true
                                    esp.box.Size = Vector2.new(sizeX, sizeY)
                                    esp.box.Position = Vector2.new(vec.X - sizeX/2, vec.Y - sizeY/2)
                                    esp.box.Color = color
                                end
                            else
                                esp.box.Visible = false
                            end
                            
                            if espConfig.tracer then
                                esp.tracer.Visible = true
                                esp.tracer.From = screenCenter
                                esp.tracer.To = Vector2.new(vec.X, vec.Y)
                                esp.tracer.Color = color
                            else
                                esp.tracer.Visible = false
                            end
                            
                            if espConfig.name then
                                esp.name.Visible = true
                                esp.name.Text = plr.Name
                                esp.name.Position = Vector2.new(vec.X, vec.Y - sizeY/2 - 15)
                                esp.name.Color = color
                            else
                                esp.name.Visible = false
                            end
                            
                            if espConfig.distance then
                                esp.distance.Visible = true
                                esp.distance.Text = math.floor((hrp.Position - camera.CFrame.Position).Magnitude) .. "m"
                                esp.distance.Position = Vector2.new(vec.X, vec.Y + sizeY/2 + 5)
                                esp.distance.Color = color
                            else
                                esp.distance.Visible = false
                            end
                        else
                            esp.box.Visible = false
                            esp.tracer.Visible = false
                            esp.name.Visible = false
                            esp.distance.Visible = false
                        end
                    elseif espDrawings[plr] then
                        for _, drawing in pairs(espDrawings[plr]) do
                            if drawing and drawing.Remove then pcall(drawing.Remove, drawing) end
                        end
                        espDrawings[plr] = nil
                    end
                end
            end
            
            -- Handle Highlight ESP
            for _, plr in pairs(players) do
                if plr ~= localPlayer and plr.Character then
                    local role = nil
                    if findMurderer() == plr then role = "Murderer"
                    elseif findSheriff() == plr then role = "Sheriff"
                    else role = "Innocent" end
                    
                    local espConfig = nil
                    if role == "Murderer" then espConfig = _G.MurdererESP
                    elseif role == "Sheriff" then espConfig = _G.SheriffESP
                    else espConfig = _G.InnocentESP end
                    
                    if espConfig and espConfig.highlight then
                        local highlight = plr.Character:FindFirstChild("PhantomHighlight")
                        if not highlight then
                            highlight = Instance.new("Highlight")
                            highlight.Name = "PhantomHighlight"
                            highlight.Adornee = plr.Character
                            highlight.FillTransparency = 0.5
                            highlight.OutlineTransparency = 0
                            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                            highlight.Parent = plr.Character
                        end
                        if role == "Murderer" then
                            highlight.FillColor = Color3.fromRGB(255, 0, 0)
                            highlight.OutlineColor = Color3.fromRGB(200, 0, 0)
                        elseif role == "Sheriff" then
                            highlight.FillColor = Color3.fromRGB(0, 0, 255)

    local function addPlayerToProcessed(player, input)
        local localPlayer = game.Players.LocalPlayer
        if not player or player == localPlayer then return end
        
        local matchesFilter = false
        local lowerInput = string.lower(input or "")
        
        if lowerInput == "all" then
            matchesFilter = true
        elseif lowerInput == "nonfriends" then
            local success, isFriend = pcall(function() return player:IsFriendsWith(localPlayer.UserId) end)
            matchesFilter = not (success and isFriend)
        elseif lowerInput == "murder" then
            matchesFilter = (findMurderer() == player)
        elseif lowerInput == "sheriff" then
            matchesFilter = (findSheriff() == player)
        else
            if string.find(string.lower(player.Name), lowerInput) or string.find(string.lower(player.DisplayName or ""), lowerInput) then
                matchesFilter = true
            end
        end
        
        if matchesFilter then processedPlayers[player] = true end
    end
    
    local function flingTarget(targetPlayer, mode, duration)
        local localPlayer = game.Players.LocalPlayer
        local character = localPlayer.Character
        local targetChar = targetPlayer.Character
        if not character or not targetChar then return end
        
        local hrp = character:FindFirstChild("HumanoidRootPart")
        local targetHrp = targetChar:FindFirstChild("HumanoidRootPart")
        if not hrp or not targetHrp then return end
        
        if mode == "Flinger" then
            local bodyVel = Instance.new("BodyVelocity", hrp)
            bodyVel.Velocity = Vector3.new(9e8, 9e8, 9e8)
            bodyVel.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            
            local startTime = tick()
            local conn
            conn = game:GetService("RunService").Heartbeat:Connect(function()
                if not _G.FlingActive or (duration and tick() - startTime > duration) then
                    conn:Disconnect()
                    bodyVel:Destroy()
                    if hrp then hrp.CFrame = originalPositions[localPlayer] or CFrame.new() end
                    return
                end
                if targetHrp and targetHrp.Parent then
                    hrp.CFrame = targetHrp.CFrame
                    targetHrp.Velocity = Vector3.new(math.random(-5000, 5000), 10000, math.random(-5000, 5000))
                end
            end)
        elseif mode == "Shhhlol" then
            local conn
            conn = game:GetService("RunService").Heartbeat:Connect(function()
                if not _G.FlingActive then conn:Disconnect() return end
                if targetHrp and targetHrp.Parent then
                    hrp.CFrame = targetHrp.CFrame
                    hrp.RotVelocity = Vector3.new(9e8, 9e8, 9e8)
                end
            end)
        elseif mode == "Yeet" then
            local thrust = Instance.new("BodyThrust", hrp)
            thrust.Force = Vector3.new(9999, 9999, 9999)
            local startTime = tick()
            local conn
            conn = game:GetService("RunService").Heartbeat:Connect(function()
                if not _G.FlingActive or (duration and tick() - startTime > duration) then
                    conn:Disconnect()
                    thrust:Destroy()
                    return
                end
                if targetHrp and targetHrp.Parent then
                    hrp.CFrame = targetHrp.CFrame
                    thrust.Location = targetHrp.Position
                end
            end)
        end
    end
    
    local function startFling()
        if not _G.FlingActive then return end
        local players = {}
        for player, _ in pairs(processedPlayers) do
            if player and player.Character then table.insert(players, player) end
        end
        
        for _, player in ipairs(players) do
            if not _G.FlingActive then break end
            local duration = (_G.FlingTarget == "all" or _G.FlingTarget == "nonfriends") and 1.5 or nil
            flingTarget(player, _G.FlingMode, duration)
            if _G.FlingTarget == "all" or _G.FlingTarget == "nonfriends" then task.wait(1.5) end
        end
        
        if _G.FlingActive then task.wait() startFling() end
    end

    -- Auto Farm Functions (From Rise)
    -- =============================================
    local function autoFarmLoop()
        while _G.AutoFarmEnabled do
            local coins = findCoins()
            local player = game.Players.LocalPlayer
            if #coins > 0 and player.Character then
                local hrp = player.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local closest = nil
                    local closestDist = math.huge
                    for _, coin in pairs(coins) do
                        local dist = (hrp.Position - coin.Position).Magnitude
                        if dist < closestDist then
                            closestDist = dist
                            closest = coin
                        end
                    end
                    if closest then
                        if _G.AutoFarmMode == "Teleport" then
                            hrp.CFrame = CFrame.new(closest.Position + Vector3.new(0, 3, 0))
                            task.wait(_G.TeleportDelay)
                        elseif _G.AutoFarmMode == "Smooth" then
                            local startPos = hrp.Position
                            local endPos = closest.Position + Vector3.new(0, 3, 0)
                            local distance = (endPos - startPos).Magnitude
                            local duration = distance / _G.SmoothSpeed
                            local startTime = tick()
                            while tick() - startTime < duration and _G.AutoFarmEnabled and closest.Parent do
                                local progress = (tick() - startTime) / duration
                                hrp.CFrame = CFrame.new(startPos:Lerp(endPos, progress))
                                task.wait()
                            end
                        else -- Walk
                            local humanoid = player.Character:FindFirstChildWhichIsA("Humanoid")
                            if humanoid then
                                humanoid.WalkSpeed = 16
                                humanoid:MoveTo(closest.Position)
                                task.wait(2)
                            end
                        end
                        firetouchinterest(hrp, closest, 0)
                        firetouchinterest(hrp, closest, 1)
                        task.wait(0.5)
                    end
                end
            end
            task.wait(_G.CoinCheckInterval)
        end
    end

            espConnections.fly = RunService.RenderStepped:Connect(function()
                if not _G.FlyEnabled or not RootPart or not RootPart.Parent then return end
                
                local moveDirection = Humanoid.MoveDirection
                local targetVelocity = Vector3.new(0, 0.1, 0)
                
                if moveDirection.Magnitude > 0 then
                    local cameraCFrame = Camera.CFrame
                    targetVelocity = moveDirection * _G.FlySpeedValue
                    
                    if math.abs(cameraCFrame.LookVector.Y) > 0.1 then
                        targetVelocity = Vector3.new(targetVelocity.X, cameraCFrame.LookVector.Y * _G.FlySpeedValue, targetVelocity.Z)
                    end

                BodyGyro.CFrame = BodyGyro.CFrame:Lerp(Camera.CFrame, 0.15)
            end)

                espConnections.fly:Disconnect()
                espConnections.fly = nil

            if BodyGyro then BodyGyro:Destroy() end
