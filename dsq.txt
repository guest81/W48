    local function findMurderer()
        local player = game.Players.LocalPlayer
        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= player then
                local hasKnife = (p.Character and p.Character:FindFirstChild("Knife")) or (p.Backpack and p.Backpack:FindFirstChild("Knife"))
                if hasKnife then return p end
            end
        end
        return nil
    end
    
    local function findSheriff()
        local player = game.Players.LocalPlayer
        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= player then
                local hasGun = (p.Character and p.Character:FindFirstChild("Gun")) or (p.Backpack and p.Backpack:FindFirstChild("Gun"))
                if hasGun then return p end
            end
        end
        return nil
    end
    
    local function findDroppedGun()
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("BasePart") and obj.Name == "GunDrop" then
                return obj
            end
        end
        return nil
    end
    
    local function findCoins()
        local coins = {}
        for _, containerName in ipairs(CoinContainers) do
            local container = workspace:FindFirstChild(containerName)
            if container then
                local coinContainer = (containerName == "Lobby") and container or container:FindFirstChild("CoinContainer")
                if coinContainer then
                    for _, coin in ipairs(coinContainer:GetChildren()) do
                        if coin:IsA("BasePart") and coin.Parent then
                            table.insert(coins, coin)
                        end
                    end
                end
            end
        end
        return coins
    end
    
    local function findNearestCoin()
        local player = game.Players.LocalPlayer
        local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
        if not hrp then return nil end
        local closest, shortest = nil, math.huge
        for _, coin in pairs(findCoins()) do
            local dist = (hrp.Position - coin.Position).Magnitude
            if dist < shortest then
                shortest = dist
                closest = coin
            end
        end
        return closest
    end

        end
    end

    local function cleanupDrawingTable(drawingTable)
        if not drawingTable then return end

                for _, line in ipairs(drawing) do

    end

    local function draw3DBox(esp, cf, pos, cam, boxColor, boxSize)
        if not cf or not cam then return end
        boxSize = boxSize or Vector3.new(4, 5, 3)
        local size = boxSize
        local offsets = {
            Vector3.new( size.X/2,  size.Y/2,  size.Z/2), Vector3.new( size.X/2,  size.Y/2, -size.Z/2),
            Vector3.new( size.X/2, -size.Y/2,  size.Z/2), Vector3.new( size.X/2, -size.Y/2, -size.Z/2),
            Vector3.new(-size.X/2,  size.Y/2,  size.Z/2), Vector3.new(-size.X/2,  size.Y/2, -size.Z/2),
            Vector3.new(-size.X/2, -size.Y/2,  size.Z/2), Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
        }
        local screenPoints, anyPoint = {}, false
        for i, offset in ipairs(offsets) do
            local success, vec, onScreen = pcall(function()
                local worldPos = cf * CFrame.Angles(0, math.rad(90), 0) * offset
                return cam:WorldToViewportPoint(worldPos)
            end)
            if success then
                screenPoints[i] = {pos = Vector2.new(vec.X, vec.Y), depth = vec.Z, onScreen = onScreen}
                if onScreen and vec.Z > 0 then anyPoint = true end
            end
        end
        if not esp.boxLines or #esp.boxLines == 0 then
            esp.boxLines = {}
            for i = 1, 12 do
                local line = Drawing.new("Line")
                line.Thickness = 1
                line.ZIndex = 2
                table.insert(esp.boxLines, line)
            end
        end
        local edges = {{1,2},{1,3},{1,5},{2,4},{2,6},{3,4},{3,7},{5,6},{5,7},{4,8},{6,8},{7,8}}
        for i, edge in ipairs(edges) do
            local line = esp.boxLines[i]
            if line then
                local p1, p2 = screenPoints[edge[1]], screenPoints[edge[2]]
                line.Color = boxColor or Color3.fromRGB(255,255,255)
                line.Thickness = 2
                if anyPoint and p1 and p2 and p1.depth > 0 and p2.depth > 0 then
                    line.From, line.To, line.Visible = p1.pos, p2.pos, true
                else
                    line.Visible = false
                end
            end
        end
    end

    local function updateRoleESP()
        local camera = workspace.CurrentCamera
        if not camera then return end
        local screenBottom = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y)
        local currentTargets = {Innocent = {}, Murder = {}, Sheriff = {}}
        
        for _, otherPlayer in pairs(game.Players:GetPlayers()) do
            if otherPlayer ~= game.Players.LocalPlayer and otherPlayer.Character then
                local hrp = otherPlayer.Character:FindFirstChild("HumanoidRootPart")
                local hum = otherPlayer.Character:FindFirstChild("Humanoid")
                if hrp and hum then
                    local hasKnife = otherPlayer.Character:FindFirstChild("Knife") or (otherPlayer.Backpack and otherPlayer.Backpack:FindFirstChild("Knife"))
                    local hasGun = otherPlayer.Character:FindFirstChild("Gun") or (otherPlayer.Backpack and otherPlayer.Backpack:FindFirstChild("Gun"))
                    local role = hasKnife and "Murderer" or (hasGun and "Sheriff") or "Innocent"
                    local isDead = hum.Health <= 0
                    
                    local config, elements, targetTable, color
                    if isDead then
                        config, elements, targetTable, color = _G.InnocentESP, innocentEspElements, currentTargets.Innocent, Color3.fromRGB(200,200,200)
                    elseif role == "Murderer" then
                        config, elements, targetTable, color = _G.MurdererESP, murderEspElements, currentTargets.Murder, Color3.fromRGB(255,0,0)
                    elseif role == "Sheriff" then
                        config, elements, targetTable, color = _G.SheriffESP, sheriffEspElements, currentTargets.Sheriff, Color3.fromRGB(0,0,255)
                    else
                        config, elements, targetTable, color = _G.InnocentESP, innocentEspElements, currentTargets.Innocent, Color3.fromRGB(0,255,0)
                    end
                    
                    -- Highlight
                    if config.highlight then
                        local highlight = otherPlayer.Character:FindFirstChild("PlayerHighlight")
                        if not highlight then
                            highlight = Instance.new("Highlight")
                            highlight.Name = "PlayerHighlight"
                            highlight.Adornee = otherPlayer.Character
                            highlight.FillTransparency = 0.5
                            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                            highlight.Parent = otherPlayer.Character
                        end
                        highlight.FillColor = color
                        highlight.OutlineColor = getOutlineColor(color)
                    else
                        local highlight = otherPlayer.Character:FindFirstChild("PlayerHighlight")
                        if highlight then highlight:Destroy() end
                    end
                    
                    if config.names or config.boxes or config.tracers or config.distance then
                        targetTable[otherPlayer.Character] = true
                        if not elements[otherPlayer.Character] then
                            elements[otherPlayer.Character] = createESPObject()
                            setupESPObject(elements[otherPlayer.Character])
                        end
                        local esp = elements[otherPlayer.Character]
                        local vec, onScreen = camera:WorldToViewportPoint(hrp.Position)
                        if onScreen then
                            local topY = camera:WorldToViewportPoint(hrp.Position + Vector3.new(0,3,0)).Y
                            local bottomY = camera:WorldToViewportPoint(hrp.Position - Vector3.new(0,3,0)).Y
                            local sizeY, sizeX = math.abs(bottomY - topY), math.abs(bottomY - topY)/2
                            if config.boxes then
                                if config.boxType == "2D" then
                                    esp.box.Visible = true
                                    esp.box.Size = Vector2.new(sizeX, sizeY)
                                    esp.box.Position = Vector2.new(vec.X - sizeX/2, vec.Y - sizeY/2)
                                    esp.box.Color = color
                                    for _, l in ipairs(esp.boxLines) do l.Visible = false end
                                else
                                    esp.box.Visible = false
                                    draw3DBox(esp, hrp.CFrame, hrp.Position, camera, color, Vector3.new(2,5,2))
                                end
                            else
                                esp.box.Visible = false
                                for _, l in ipairs(esp.boxLines) do l.Visible = false end
                            end
                            if config.tracers then
                                esp.tracer.Visible = true
                                esp.tracer.From, esp.tracer.To, esp.tracer.Color = screenBottom, Vector2.new(vec.X, vec.Y), color
                            else
                                esp.tracer.Visible = false
                            end
                            if config.names then
                                esp.name.Visible = true
                                esp.name.Text, esp.name.Position, esp.name.Color = otherPlayer.Name, Vector2.new(vec.X, vec.Y - sizeY/2 - 15), color
                            else
                                esp.name.Visible = false
                            end

    local function updateItemESP(config, elementTable, findFunc, itemName, color)
        local camera = workspace.CurrentCamera
        if not camera then return end
        local screenBot = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y)
        local currentItems = {}
        for _, item in ipairs(findFunc()) do
            if item and item.Parent then

                    elementTable[item] = createESPObject()

                end
                local esp = elementTable[item]
                local vec, onScreen = camera:WorldToViewportPoint(item.Position)

                        esp.box.Visible = false
                        for _, l in ipairs(esp.boxLines) do l.Visible = false end

                        esp.tracer.Visible = true
                        esp.tracer.From, esp.tracer.To, esp.tracer.Color = screenBot, Vector2.new(vec.X, vec.Y), color

                        esp.tracer.Visible = false

                        esp.name.Text, esp.name.Position, esp.name.Color = itemName, Vector2.new(vec.X, vec.Y - size/2 - 15), color

                        esp.name.Visible = false

    local function ShootMurderer()
        local player = game.Players.LocalPlayer
        if not player.Character then return false end
        local murderer = findMurderer()
        if not murderer or not murderer.Character then return false end
        local gun = player.Character:FindFirstChild("Gun") or player.Backpack:FindFirstChild("Gun")
        if not gun then return false end
        if gun.Parent ~= player.Character then
            gun.Parent = player.Character
            task.wait(0.1)
        end
        gun = player.Character:FindFirstChild("Gun")
        if gun and gun:FindFirstChild("KnifeLocal") then
            local targetPart = murderer.Character:FindFirstChild("HumanoidRootPart")
            if targetPart then
                local targetPos = targetPart.Position
                local mHum = murderer.Character:FindFirstChild("Humanoid")
                if _G.ShootPrediction ~= 0 and mHum then
                    local moveDir = mHum.MoveDirection
                    if moveDir.Magnitude > 0 then
                        targetPos = targetPos + (moveDir.Unit * (_G.ShootPrediction + (_G.PingMultiplier * 0.5)))
                    end
                end
                pcall(function()
                    gun.KnifeLocal.CreateBeam.RemoteFunction:InvokeServer(1, targetPos, "AH2")
                end)
                return true
            end
        end
        return false
    end

    local function getPlayersForFling(input)
        local player = game.Players.LocalPlayer
        local players, input = {}, string.lower(input or "")
        if input == "all" then
            for _, p in pairs(game.Players:GetPlayers()) do if p ~= player then table.insert(players, p) end end
        elseif input == "nonfriends" then
            for _, p in pairs(game.Players:GetPlayers()) do
                if p ~= player then
                    local success, isFriend = pcall(function() return p:IsFriendsWith(player.UserId) end)
                    if not (success and isFriend) then table.insert(players, p) end
                end
            end
        elseif input == "murder" then
            local m = findMurderer()
            if m and m ~= player and m.Character and IsAlive(m) then table.insert(players, m) end
        elseif input == "sheriff" then
            local s = findSheriff()
            if s and s ~= player and s.Character and IsAlive(s) then table.insert(players, s) end
        else
            for _, p in pairs(game.Players:GetPlayers()) do
                if p ~= player then
                    if string.find(string.lower(p.Name), input) or string.find(string.lower(p.DisplayName or ""), input) then
                        table.insert(players, p)
                    end
                end
            end
        end
        return players
    end
    
    local function Flinger(TargetPlayer, duration)
        local player, startTime = game.Players.LocalPlayer, tick()
        local Char, Humanoid, RootPart = player.Character, player.Character and player.Character:FindFirstChildOfClass("Humanoid"), player.Character and player.Character:FindFirstChild("Humanoid") and player.Character.Humanoid.RootPart

        if not TChar:FindFirstChildWhichIsA("BasePart") then return end

        Humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
        local function SF(BasePart)
            local Time, Angle = tick(), 0
            repeat
                if RootPart and THumanoid then
                    Angle = Angle + 100
                    RootPart.CFrame, RootPart.Velocity, RootPart.RotVelocity = CFrame.new(BasePart.Position) * CFrame.new(0,1.5,0) * CFrame.Angles(math.rad(Angle),0,0), Vector3.new(9e7,9e7*10,9e7), Vector3.new(9e8,9e8,9e8)
                    task.wait()

                    task.wait()
                else break end
            until not _G.FlingActive or BasePart.Velocity.Magnitude > 500 or tick() > Time + (duration or 2)
        end
        if TRootPart and THead then
            if (TRootPart.CFrame.p - THead.CFrame.p).Magnitude > 5 then SF(THead) else SF(TRootPart) end
        elseif TRootPart then SF(TRootPart)
        elseif THead then SF(THead) end
        BV:Destroy()

        workspace.CurrentCamera.CameraSubject = Humanoid
        repeat
            if Char and Humanoid and RootPart and getgenv().OldPos then

            end
            task.wait()
        until not _G.FlingActive or (RootPart and getgenv().OldPos and (RootPart.Position - getgenv().OldPos.p).Magnitude < 25)

        conn = game:GetService("RunService").Heartbeat:Connect(function()

    local function teleportToCoin(coin)
        local hrp = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

            hrp.CFrame = CFrame.new(coin.Position + Vector3.new(0,3,0))

    local function smoothMoveToCoin(coin)
        local hrp = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not hrp or not coin then return end
        local startPos, endPos = hrp.Position, coin.Position + Vector3.new(0,3,0)
        local duration = (endPos - startPos).Magnitude / _G.MoveSpeed
        local startTime = tick()
        while tick() - startTime < duration and _G.AutoFarmEnabled do
            if not coin.Parent then break end
            hrp.CFrame = CFrame.new(startPos:Lerp(endPos, (tick() - startTime)/duration))
            task.wait()
        end
    end
    
    local function walkToCoin(coin)
        local humanoid = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if not humanoid or not coin then return end
        humanoid.WalkSpeed = 16
        humanoid:MoveTo(coin.Position + Vector3.new(0,0,3))
        local start = tick()
        while _G.AutoFarmEnabled and humanoid.MoveDirection.Magnitude > 0 and tick() - start < 10 do task.wait(0.5) end
    end
    
    local function collectCoin(coin)
        local hrp = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hrp and coin and coin.Parent then
            firetouchinterest(hrp, coin, 0)
            firetouchinterest(hrp, coin, 1)
            task.wait(1)
        end
    end

    local function setupFly()
        local player, RunService, Camera = game.Players.LocalPlayer, game:GetService("RunService"), workspace.CurrentCamera
        if _G.FlyEnabled then
            if BodyVelocity then BodyVelocity:Destroy() end
            if BodyGyro then BodyGyro:Destroy() end
            local Char, RootPart, Humanoid = player.Character, player.Character and player.Character:FindFirstChild("HumanoidRootPart"), player.Character and player.Character:FindFirstChild("Humanoid")
            if not (Char and RootPart and Humanoid) then return end
            BodyVelocity = Instance.new("BodyVelocity", RootPart)
            BodyVelocity.MaxForce, BodyVelocity.Velocity = Vector3.new(math.huge, math.huge, math.huge), Vector3.new(0,0.1,0)
            BodyGyro = Instance.new("BodyGyro", RootPart)
            BodyGyro.MaxTorque, BodyGyro.P, BodyGyro.D, BodyGyro.CFrame = Vector3.new(math.huge, math.huge, math.huge), 10000, 500, RootPart.CFrame
            Humanoid.PlatformStand = true
            if espConnections.fly then espConnections.fly:Disconnect() end
            espConnections.fly = RunService.RenderStepped:Connect(function()
                if not _G.FlyEnabled or not RootPart or not RootPart.Parent then return end
                local moveDir = Humanoid.MoveDirection
                local targetVel = Vector3.new(0,0.1,0)
                if moveDir.Magnitude > 0 then
                    local camCF = Camera.CFrame
                    targetVel = moveDir * _G.FlySpeedValue
                    if math.abs(camCF.LookVector.Y) > 0.1 then
                        targetVel = Vector3.new(targetVel.X, camCF.LookVector.Y * _G.FlySpeedValue, targetVel.Z)
                    end
                end
                BodyVelocity.Velocity = BodyVelocity.Velocity:Lerp(targetVel, 0.15)
                BodyGyro.CFrame = BodyGyro.CFrame:Lerp(Camera.CFrame, 0.15)
            end)
        else
            if espConnections.fly then espConnections.fly:Disconnect(); espConnections.fly = nil end
            if BodyVelocity then BodyVelocity:Destroy() end
            if BodyGyro then BodyGyro:Destroy() end
            BodyVelocity, BodyGyro = nil, nil
            local Char = player.Character
            if Char and Char:FindFirstChild("Humanoid") then Char.Humanoid.PlatformStand = false end
        end
    end
